CREATE TABLE `Users` (
  `user_ID` int NOT NULL AUTO_INCREMENT,
  `first_name` varchar(50),
  `last_name` varchar(50),
  `email` varchar(50),
  `password` varchar(50),
  PRIMARY KEY (`user_ID`),
  KEY `Key` (`first_name`, `last_name`, `email`, `password`)
);

CREATE TABLE `Instructors` (
  `instructor_ID` int NOT NULL AUTO_INCREMENT,
  `user_ID` int,
  `description` longtext,
  `num_of_students` int,
  PRIMARY KEY (`instructor_ID`),
  FOREIGN KEY (`user_ID`) REFERENCES Users(user_ID),
  KEY `Key` (`num_of_students`)
);


CREATE TABLE `Courses` (
  `course_ID` int NOT NULL AUTO_INCREMENT,
  `instructor_ID` int,
  `title` varchar(50),
  `genre` varchar(50),
  `price` double,
  `rating` float,
  `num_of_enrolments` int,
  `description` longtext,
  `publish_date` datetime,
  PRIMARY KEY (`course_ID`),
  FOREIGN Key (`instructor_ID`) REFERENCES Instructors(instructor_ID),
  KEY `Key` (`title`, `genre`, `price`, `rating`, `num_of_enrolments`, `publish_date`)
);

CREATE TABLE `Syllabuses` (
  `syllabus_ID` int NOT NULL AUTO_INCREMENT,
  `course_ID` int,
  `syllabus_Name` varchar(50),
  PRIMARY KEY (`syllabus_ID`),
  FOREIGN KEY (`course_ID`) REFERENCES Courses(course_ID),
  KEY `Key` (`syllabus_name`)
);

CREATE TABLE `Orders` (
  `order_ID` int NOT NULL AUTO_INCREMENT,
  `course_ID` int,
  `user_ID` int,
  `order_date` datetime,
  PRIMARY KEY (`order_ID`),
  FOREIGN Key (`course_ID`) REFERENCES Courses(course_ID),
    FOREIGN Key (`user_ID`) REFERENCES Users(user_ID),
  KEY `Key` (`order_date`)
);

CREATE TABLE `Contents` (
  `content_ID` int NOT NULL AUTO_INCREMENT,
  `syllabus_ID` int,
  `title` varchar(50),
  `description` longtext,
  `video` BLOB,
  PRIMARY KEY (`content_ID`),
  FOREIGN KEY (`syllabus_ID`) REFERENCES Syllabuses(syllabus_ID),
  KEY `Key` (`title`)
);
